All Country Table   ====>     CREATE TABLE `country_state_city`.`countries` ( `id` INT(255) NOT NULL AUTO_INCREMENT , `name` VARCHAR(255) NULL , `iso3` VARCHAR(255) NULL , `iso2` VARCHAR(255) NULL , `phone_code` VARCHAR(255) NULL , `capital` VARCHAR(255) NULL , `currency` VARCHAR(255) NULL , `currency_symbol` VARCHAR(255) NULL , `native` VARCHAR(255) NULL , `region` VARCHAR(255) NULL , `subregion` VARCHAR(255) NULL , `latitude` VARCHAR(255) NULL , `longitude` VARCHAR(255) NULL , `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
